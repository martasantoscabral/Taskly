    nano .env

    DATABASE_URL="postgresql://taskly_user:1234@localhost:5432/taskly_db"

##Instalar dependencias##
1. express
    npm install express dotenv
2.node_modules  
    npm install -D typescript tsx @types/node @types/express
3.prisma
    npm install prisma@6 @prisma/client@7
4. adapter
    npm install @prisma/adapter-pg pg
5. zod
    npm install zod swagger-ui-express swagger-jsdoc
6. swagger
    npm install -D @types/swagger-ui-express @types/swagger-jsdoc
7. helmet
    npm install helmet cors express-rate-limit jsonwebtoken bcryptjs
8.
    npm install -D @types/jsonwebtoken @types/bcryptjs
